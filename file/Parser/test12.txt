//一个人从地面开始可以一次上一级台阶，也可以一次上两级台阶
//最多可以一次上三级台阶，请问，上到15级有多少中上楼梯的方法
int f(int);
main()
{
   int m;
   m = read();
   write(f(m));

}

int f(int n) {
   int a,b,c;
   if(n==1) {
		return 1;
	}
   else if(n==2) {
		return 2;
	}
   else if(n==3) {
		return 4;
	}
    a = f(n - 1);
    b = f(n - 2);
    c = f(n - 3);
    return a + b + c;
 }
